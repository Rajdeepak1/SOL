func ( greed , cookies )
{
    n = greed.size() ; m= cookies.size();
    int l=0; int r=0;

    sort(greed.begin(),greed.end());
    sort(cookies.begin(),cookies.end());

    while(l<m)
    {
        if( greed[r] <= cookies[l]) r = r+1;
        l=l+1;
    }
    return r;
}
